tail -fn 100 /data/logs/applogs/dongdao-api-mapi-cp/daodao-api-mapi-cp.log
# 查看api-cp访问路径
tail -fn 100 /data/logs/applogs/dongdao-api-mapi-cp/localhost_access_log.*
